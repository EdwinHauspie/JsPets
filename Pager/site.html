Pager

Requires

  • jQuery

Usage

var $myPager = $.createPager({
    pageIndex: 0,
    pageSize: 10,
    rowCount: 433,
    text: {
        next: '›',
        previous: '‹',
        first: '‹‹',
        last: '››',
        statusFormat: '{0} / {1}',
        pageSizeFormat: 'Page size {0}'
    }
});

$myPager
    .on('onPageChanged', your_event_handler)
    .on('onPageSizeChanged', your_event_handler);

$('#myPagerWrapper').append($myPager);
